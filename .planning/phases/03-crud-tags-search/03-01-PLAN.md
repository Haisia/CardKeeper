# Phase 3 Plan 01: CardListScreen + CardListViewModel

## Objective

Create a scrollable card list showing thumbnails, names, and company names, with a floating action button (FAB) to start a new scan. The list will display saved cards from Room in real-time using a reactive Flow, and tapping a card navigates to the CardDetailScreen.

**Goal**: Enable users to browse all saved cards and initiate a new scan from the card list.

**Requirements**: BROWSE-01 (see ROADMAP.md).

---

## Background

**Why a dedicated CardListScreen?**
- Separate from ScanScreen to make the card browsing experience focused and consistent
- Reuses existing components: Coil for image loading, LazyColumn for scrolling
- Uses `CardWithTags` POJO (from Phase 1 plan 01-02) to include tags in the list
- Provides clear navigation: tap card → detail, FAB → scan

**Architecture**:
- ViewModel subscribes to `CardRepository.getAllCards()` (Flow)
- Composable observes `Flow<List<CardWithTags>>` and renders LazyColumn
- Each list item shows: thumbnail, name, company (collapsed for long names)
- FAB positioned at bottom-right, fixed to bottom navigation
- Empty state shown when list is empty (Cards 0)

package com.cardkeeper.ui.card

import androidx.lifecycle.ViewModel
import androidx.lifecycle.viewModelScope
import com.cardkeeper.data.db.CardWithTags
import com.cardkeeper.domain.repository.CardRepository
import dagger.hilt.android.lifecycle.HiltViewModel
import kotlinx.coroutines.flow.MutableStateFlow
import kotlinx.coroutines.flow.StateFlow
import kotlinx.coroutines.flow.asStateFlow
import kotlinx.coroutines.launch
import javax.inject.Inject

// ----------------------------------------------------------------
// UI state models
// ----------------------------------------------------------------

data class CardListUiState(
    val cards: List<CardWithTags> = emptyList(),
    val isLoading: Boolean = false,
    val error: String? = null
)

// ----------------------------------------------------------------
// ViewModel
// ----------------------------------------------------------------

@HiltViewModel
class CardListViewModel @Inject constructor(
    private val cardRepository: CardRepository
) : ViewModel() {

    private val _uiState = MutableStateFlow(CardListUiState())
    val uiState: StateFlow<CardListUiState> = _uiState.asStateFlow()

    init {
        loadCards()
    }

    /**
     * Load all cards from Room and update uiState.
     * Error is captured and stored; loading state is transient.
     */
    private fun loadCards() {
        _uiState.value = _uiState.value.copy(isLoading = true, error = null)
        viewModelScope.launch {
            try {
                cardRepository.getAllCards()
                    .collect { cards ->
                        _uiState.value = _uiState.value.copy(
                            cards = cards,
                            isLoading = false,
                            error = null
                        )
                    }
            } catch (e: Exception) {
                _uiState.value = _uiState.value.copy(
                    isLoading = false,
                    error = e.message ?: "Failed to load cards"
                )
            }
        }
    }

    /**
     * Refresh the card list by restarting the Flow subscription.
     * Called when user pulls to refresh.
     */
    fun refresh() {
        loadCards()
    }
}
```

**Key design decisions**:
- `_uiState` holds the reactive state: `cards`, `isLoading`, `error`
- `isLoading` is set to `true` at the start of `loadCards()` then cleared in the collect block
- Error is stored in `_uiState` so the UI can display it (e.g., snackbar)
- `refresh()` re-triggers the Flow subscription by calling `loadCards()` again

```

**Acceptance criteria**:

- `CardListViewModel.kt` contains `fun loadCards()` that subscribes to `cardRepository.getAllCards()`

- `CardListViewModel.kt` contains `private val _uiState = MutableStateFlow(CardListUiState())`

@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun CardListScreen(
    viewModel: CardListViewModel,
    onCardClick: (CardWithTags) -> Unit,
    onScan: () -> Unit,
    onBack: () -> Unit
) {
    val uiState by viewModel.uiState.collectAsState()
    val snackbarHostState = remember { SnackbarHostState() }

    // Show error in snackbar when uiState.error is not null
    LaunchedEffect(uiState.error) {
        uiState.error?.let { message ->
            snackbarHostState.showSnackbar(message)
        }
    }

    Scaffold(
        topBar = {
            TopAppBar(
                title = { Text("My Cards") },
                navigationIcon = {
                    IconButton(onClick = onBack) {
                        Icon(
                            imageVector = Icons.Default.ArrowBack,
                            contentDescription = "Back"
                        )
                    }
                }
            )
        },
        floatingActionButton = {
            FloatingActionButton(
                onClick = onScan,
                modifier = Modifier.padding(bottom = 16.dp, end = 16.dp)
            ) {
                Icon(Icons.Default.Add, contentDescription = "Scan")
            }
        },
        snackbarHost = { SnackbarHost(snackbarHostState) }
    ) { innerPadding ->
        Column(
            modifier = Modifier
                .fillMaxSize()
                .padding(innerPadding)
        ) {
            if (uiState.isLoading) {
                // Show loading indicator while loading cards
                Box(
                    modifier = Modifier
                        .fillMaxSize()
                        .padding(16.dp),
                    contentAlignment = Alignment.Center
                ) {
                    CircularProgressIndicator()
                }
            } else if (uiState.cards.isEmpty()) {
                // Show empty state
                Box(
                    modifier = Modifier
                        .fillMaxSize()
                        .padding(16.dp),
                    contentAlignment = Alignment.Center
                ) {
                    Column(
                        horizontalAlignment = Alignment.CenterHorizontally,
                        verticalArrangement = Arrangement.spacedBy(8.dp)
                    ) {
                        Text(
                            "No cards yet",
                            style = MaterialTheme.typography.headlineMedium
                        )
                        Text(
                            "Tap the + button to scan your first card",
                            style = MaterialTheme.typography.bodyLarge,
                            color = MaterialTheme.colorScheme.onSurfaceVariant
                        )
                    }
                }
            } else {
                // Show card list
                LazyColumn(
                    modifier = Modifier.fillMaxSize(),
                    contentPadding = PaddingValues(16.dp),
                    verticalArrangement = Arrangement.spacedBy(8.dp)
                ) {
                    items(uiState.cards, key = { it.card.id }) { card ->
                        CardListItem(
                            card = card,
                            onClick = { onCardClick(card) }
                        )
                    }
                }
            }
        }
    }
}

@Composable
private fun CardListItem(
    card: CardWithTags,
    onClick: () -> Unit
) {
    ListItem(
        modifier = Modifier
            .fillMaxWidth()
            .clickable(onClick = onClick),
        leadingContent = {
            AsyncImage(
                model = ImageRequest.Builder(LocalContext.current)
                    .data(card.card.imagePath)
                    .diskCacheKey(card.card.imagePath)
                    .size(ImageRequest.Size.ORIGINAL)
                    .build(),
                contentDescription = "Card image",
                modifier = Modifier.size(64.dp),
                contentScale = ContentScale.Crop
            )
        },
        title = {
            Text(
                text = card.card.name,
                style = MaterialTheme.typography.titleMedium,
                maxLines = 1,
                overflow = TextOverflow.Ellipsis
            )
        },
        supportingText = {
            Text(
                text = card.card.company,
                style = MaterialTheme.typography.bodyMedium,
                maxLines = 1,
                overflow = TextOverflow.Ellipsis
            )
        },
        trailingContent = {
            // Show first tag if available
            card.tags.firstOrNull()?.let { tag ->
                Text(
                    text = tag.name,
                    style = MaterialTheme.typography.labelSmall,
                    color = MaterialTheme.colorScheme.primary
                )
            }
        }
    )
}
