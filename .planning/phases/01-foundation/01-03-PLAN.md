    ```kotlin
    class CardRepositoryImpl @Inject constructor(
        private val cardDao: CardDao
    ) : CardRepository {
        override fun getAllCards() = cardDao.getAllCardsWithTags()
        override fun getCardById(cardId: Long) = cardDao.getCardWithTagsById(cardId)
        override fun searchCards(query: String) = cardDao.searchCards("%$query%")
        override suspend fun insertCard(card: CardEntity) = cardDao.insertCard(card)
        override suspend fun updateCard(card: CardEntity) = cardDao.updateCard(card)
        override suspend fun deleteCard(card: CardEntity) = cardDao.deleteCard(card)
    }
    ```
    Note: `searchCards` wraps the query with `%` wildcards for LIKE pattern matching.

    **data/repository/TagRepositoryImpl.kt** (package: `com.cardkeeper.data.repository`):
    ```kotlin
    class TagRepositoryImpl @Inject constructor(
        private val tagDao: TagDao
    ) : TagRepository {
        override fun getAllTags() = tagDao.getAllTags()
        override suspend fun insertTag(tag: TagEntity) = tagDao.insertTag(tag)
        override suspend fun deleteTag(tag: TagEntity) = tagDao.deleteTag(tag)
        override suspend fun setTagsForCard(cardId: Long, tagIds: List<Long>) {
            tagDao.deleteTagsForCard(cardId)
            tagIds.forEach { tagId ->
                tagDao.insertCrossRef(CardTagCrossRef(cardId = cardId, tagId = tagId))
            }
        }
    }
    ```

  <action>
    Create four Hilt modules in `app/src/main/kotlin/com/cardkeeper/di/`. Package: `com.cardkeeper.di`. All modules installed in `SingletonComponent`.

    **DatabaseModule.kt** — Uses `@Provides` (Room objects are not constructor-injectable):
    ```kotlin
    @Module
    @InstallIn(SingletonComponent::class)
    object DatabaseModule {
        @Singleton
        @Provides
        fun provideDatabase(@ApplicationContext context: Context): AppDatabase =
            AppDatabase.getInstance(context)

        @Provides
        fun provideCardDao(db: AppDatabase): CardDao = db.cardDao()

        @Provides
        fun provideTagDao(db: AppDatabase): TagDao = db.tagDao()
    }
    ```
    Imports: `dagger.Module`, `dagger.Provides`, `dagger.hilt.InstallIn`, `dagger.hilt.android.qualifiers.ApplicationContext`, `dagger.hilt.components.SingletonComponent`, `javax.inject.Singleton`, `android.content.Context`, and the db classes.

    **RepositoryModule.kt** — Uses `@Binds` (implementations have @Inject constructors):
    ```kotlin
    @Module
    @InstallIn(SingletonComponent::class)
    abstract class RepositoryModule {
        @Singleton
        @Binds
        abstract fun bindCardRepository(impl: CardRepositoryImpl): CardRepository

        @Singleton
        @Binds
        abstract fun bindTagRepository(impl: TagRepositoryImpl): TagRepository
    }
    ```
    CRITICAL: This is an `abstract class`, not an `object`. `@Binds` requires abstract methods.

    **StorageModule.kt** — Provides ImageStorageDataSource (stub, real impl Phase 2):

    object StorageModule {
